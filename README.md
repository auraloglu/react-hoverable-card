## Install

```bash
npm install --save react-hoverable-card
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-hoverable-card'
import 'react-hoverable-card/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
